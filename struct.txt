
struct Person {
    string name;
    int age;
    bool isMale;
};

int main() {
    // Create a Person object
    Person john;

    // Set the object's properties
    john.name = "John";
    john.age = 30;
    john.isMale = true;

    // Print the object's properties
    cout << "Name: " << john.name << endl;
    cout << "Age: " << john.age << endl;
    cout << "Is Male: " << (john.isMale ? "Yes" : "No") << endl;

    return 0;
